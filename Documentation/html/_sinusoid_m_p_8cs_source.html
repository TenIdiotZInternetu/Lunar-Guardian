97 private Vector3 GetNextPosition(float time)
98 {
99 float currentSpeed = speed + acceleration * time;
100 float currentAmplitude = amplitude + amplitudeChange * time;
101 float currentFrequency = frequency + frequencyChange * time;
102 float currentPhase = (phase + phaseShift * Time.time) % 360;
103
104 float distance = currentSpeed * time;
105
106 Vector3 axisPosition = !staysInLine ?
107 GetAxisPosition(distance) :
108 _origin;
109
110 Vector3 sinusoidPosition = GetSinusoidPosition(distance, currentAmplitude, currentFrequency, currentPhase);

123 {
124 phase *= Mathf.Deg2Rad;
125 float sineValue = amplitude * Mathf.Sin(frequency * t + phase);
