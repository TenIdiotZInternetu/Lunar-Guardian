346 void ProcessBezierPathInput (Event e) {
347 // Find which handle mouse is over. Start by looking at previous handle index first, as most likely to still be closest to mouse
348 int previousMouseOverHandleIndex = (mouseOverHandleIndex == -1) ? 0 : mouseOverHandleIndex;
349 mouseOverHandleIndex = -1;
350 for (int i = 0; i < bezierPath.NumPoints; i += 3) {
351
352 int handleIndex = (previousMouseOverHandleIndex + i) % bezierPath.NumPoints;
353 float handleRadius = GetHandleDiameter (globalDisplaySettings.anchorSize * data.bezierHandleScale, bezierPath[handleIndex]) / 2f;
354 Vector3 pos = MathUtility.TransformPoint (bezierPath[handleIndex], creator.transform, bezierPath.Space);
355 float dst = HandleUtility.DistanceToCircle (pos, handleRadius);
356 if (dst == 0) {
357 mouseOverHandleIndex = handleIndex;
358 break;
359 }
360 }
361
362 // Shift-left click (when mouse not over a handle) to split or add segment
363 if (mouseOverHandleIndex == -1) {
364 if (e.type == EventType.MouseDown && e.button == 0 && e.shift) {
365 UpdatePathMouseInfo ();
366 // Insert point along selected segment
367 if (selectedSegmentIndex != -1 && selectedSegmentIndex < bezierPath.NumSegments) {
368 Vector3 newPathPoint = pathMouseInfo.closestWorldPointToMouse;
369 newPathPoint = MathUtility.InverseTransformPoint (newPathPoint, creator.transform, bezierPath.Space);
370 Undo.RecordObject (creator, "Split segment");
371 bezierPath.SplitSegment (newPathPoint, selectedSegmentIndex, pathMouseInfo.timeOnBezierSegment);
372 }
373 // If path is not a closed loop, add new point on to the end of the path
374 else if (!bezierPath.IsClosed) {
375 // insert new point at same dst from scene camera as the point that comes before it (for a 3d path)
376 float dstCamToEndpoint = (Camera.current.transform.position - bezierPath[bezierPath.NumPoints - 1]).magnitude;
377 Vector3 newPathPoint = MouseUtility.GetMouseWorldPosition (bezierPath.Space, dstCamToEndpoint);
378 newPathPoint = MathUtility.InverseTransformPoint (newPathPoint, creator.transform, bezierPath.Space);
379
380 Undo.RecordObject (creator, "Add segment");
381 if (e.control || e.command) {
382 bezierPath.AddSegmentToStart (newPathPoint);
383 } else {
384 bezierPath.AddSegmentToEnd (newPathPoint);
385 }
386
387 }
388
389 }
390 }
391
392 // Control click or backspace/delete to remove point
393 if (e.keyCode == KeyCode.Backspace || e.keyCode == KeyCode.Delete || ((e.control || e.command) && e.type == EventType.MouseDown && e.button == 0)) {
394
395 if (mouseOverHandleIndex != -1) {
396 Undo.RecordObject (creator, "Delete segment");
397 bezierPath.DeleteSegment (mouseOverHandleIndex);
398 if (mouseOverHandleIndex == handleIndexToDisplayAsTransform) {
399 handleIndexToDisplayAsTransform = -1;
400 }
401 mouseOverHandleIndex = -1;
402 Repaint ();
403 }
404 }
405
406 // Holding shift and moving mouse (but mouse not over a handle/dragging a handle)
407 if (draggingHandleIndex == -1 && mouseOverHandleIndex == -1) {
408 bool shiftDown = e.shift && !shiftLastFrame;
409 if (shiftDown || ((e.type == EventType.MouseMove || e.type == EventType.MouseDrag) && e.shift)) {
410
411 UpdatePathMouseInfo ();
412
413 if (pathMouseInfo.mouseDstToLine < segmentSelectDistanceThreshold) {
414 if (pathMouseInfo.closestSegmentIndex != selectedSegmentIndex) {
415 selectedSegmentIndex = pathMouseInfo.closestSegmentIndex;
416 HandleUtility.Repaint ();
417 }
418 } else {
419 selectedSegmentIndex = -1;
420 HandleUtility.Repaint ();
421 }
422
423 }
424 }
425
426 shiftLastFrame = e.shift;
427
428 }

430 void DrawBezierPathSceneEditor () {
431
432 bool displayControlPoints = data.displayControlPoints && (bezierPath.ControlPointMode != BezierPath.ControlMode.Automatic || !globalDisplaySettings.hideAutoControls);
433 Bounds bounds = bezierPath.CalculateBoundsWithTransform (creator.transform);
434
435 if (Event.current.type == EventType.Repaint) {
436 for (int i = 0; i < bezierPath.NumSegments; i++) {
437 Vector3[] points = bezierPath.GetPointsInSegment (i);
438 for (int j = 0; j < points.Length; j++) {
439 points[j] = MathUtility.TransformPoint (points[j], creator.transform, bezierPath.Space);
440 }
441
442 if (data.showPerSegmentBounds) {
443 Bounds segmentBounds = CubicBezierUtility.CalculateSegmentBounds (points[0], points[1], points[2], points[3]);
444 Handles.color = globalDisplaySettings.segmentBounds;
445 Handles.DrawWireCube (segmentBounds.center, segmentBounds.size);
446 }
447
448 // Draw lines between control points
449 if (displayControlPoints) {
450 Handles.color = (bezierPath.ControlPointMode == BezierPath.ControlMode.Automatic) ? globalDisplaySettings.handleDisabled : globalDisplaySettings.controlLine;
451 Handles.DrawLine (points[1], points[0]);
452 Handles.DrawLine (points[2], points[3]);
453 }
454
455 // Draw path
456 bool highlightSegment = (i == selectedSegmentIndex && Event.current.shift && draggingHandleIndex == -1 && mouseOverHandleIndex == -1);
457 Color segmentCol = (highlightSegment) ? globalDisplaySettings.highlightedPath : globalDisplaySettings.bezierPath;
458 Handles.DrawBezier (points[0], points[3], points[1], points[2], segmentCol, null, 2);
459 }
460
461 if (data.showPathBounds) {
462 Handles.color = globalDisplaySettings.bounds;
463 Handles.DrawWireCube (bounds.center, bounds.size);
464 }
465
466 // Draw normals
467 if (data.showNormals) {
468 if (!hasUpdatedNormalsVertexPath) {
469 normalsVertexPath = new VertexPath (bezierPath, creator.transform, normalsSpacing);
470 hasUpdatedNormalsVertexPath = true;
471 }
472
473 if (editingNormalsOld != data.showNormals) {
474 editingNormalsOld = data.showNormals;
475 Repaint ();
476 }
477
478 Vector3[] normalLines = new Vector3[normalsVertexPath.NumPoints * 2];
479 Handles.color = globalDisplaySettings.normals;
480 for (int i = 0; i < normalsVertexPath.NumPoints; i++) {
481 normalLines[i * 2] = normalsVertexPath.GetPoint (i);
482 normalLines[i * 2 + 1] = normalsVertexPath.GetPoint (i) + normalsVertexPath.GetNormal (i) * globalDisplaySettings.normalsLength;
483 }
484 Handles.DrawLines (normalLines);
485 }
486 }
487
488 if (data.displayAnchorPoints) {
489 for (int i = 0; i < bezierPath.NumPoints; i += 3) {
490 DrawHandle (i);
491 }
492 }
493 if (displayControlPoints) {
494 for (int i = 1; i < bezierPath.NumPoints - 1; i += 3) {
495 DrawHandle (i);
496 DrawHandle (i + 1);
497 }
498 }

501 void DrawHandle (int i) {
502 Vector3 handlePosition = MathUtility.TransformPoint (bezierPath[i], creator.transform, bezierPath.Space);
503
504 float anchorHandleSize = GetHandleDiameter (globalDisplaySettings.anchorSize * data.bezierHandleScale, bezierPath[i]);
505 float controlHandleSize = GetHandleDiameter (globalDisplaySettings.controlSize * data.bezierHandleScale, bezierPath[i]);
506
507 bool isAnchorPoint = i % 3 == 0;
508 bool isInteractive = isAnchorPoint || bezierPath.ControlPointMode != BezierPath.ControlMode.Automatic;
509 float handleSize = (isAnchorPoint) ? anchorHandleSize : controlHandleSize;
510 bool doTransformHandle = i == handleIndexToDisplayAsTransform;
511
512 PathHandle.HandleColours handleColours = (isAnchorPoint) ? splineAnchorColours : splineControlColours;
513 if (i == handleIndexToDisplayAsTransform) {
514 handleColours.defaultColour = (isAnchorPoint) ? globalDisplaySettings.anchorSelected : globalDisplaySettings.controlSelected;
515 }
516 var cap = capFunctions[(isAnchorPoint) ? globalDisplaySettings.anchorShape : globalDisplaySettings.controlShape];
517 PathHandle.HandleInputType handleInputType;
518 handlePosition = PathHandle.DrawHandle (handlePosition, bezierPath.Space, isInteractive, handleSize, cap, handleColours, out handleInputType, i);
519
520 if (doTransformHandle) {
521 // Show normals rotate tool
522 if (data.showNormals && Tools.current == Tool.Rotate && isAnchorPoint && bezierPath.Space == PathSpace.xyz) {
523 Handles.color = handlesStartCol;
524
525 int attachedControlIndex = (i == bezierPath.NumPoints - 1) ? i - 1 : i + 1;
526 Vector3 dir = (bezierPath[attachedControlIndex] - handlePosition).normalized;
527 float handleRotOffset = (360 + bezierPath.GlobalNormalsAngle) % 360;
528 anchorAngleHandle.radius = handleSize * 3;
529 anchorAngleHandle.angle = handleRotOffset + bezierPath.GetAnchorNormalAngle (i / 3);
530 Vector3 handleDirection = Vector3.Cross (dir, Vector3.up);
531 Matrix4x4 handleMatrix = Matrix4x4.TRS (
532 handlePosition,
533 Quaternion.LookRotation (handleDirection, dir),
534 Vector3.one
535 );
536
537 using (new Handles.DrawingScope (handleMatrix)) {
538 // draw the handle
539 EditorGUI.BeginChangeCheck ();
540 anchorAngleHandle.DrawHandle ();
541 if (EditorGUI.EndChangeCheck ()) {
542 Undo.RecordObject (creator, "Set angle");
543 bezierPath.SetAnchorNormalAngle (i / 3, anchorAngleHandle.angle - handleRotOffset);
544 }
545 }
546
547 } else {
548 handlePosition = Handles.DoPositionHandle (handlePosition, Quaternion.identity);
549 }
550
551 }
552
553 switch (handleInputType) {
554 case PathHandle.HandleInputType.LMBDrag:
555 draggingHandleIndex = i;
556 handleIndexToDisplayAsTransform = -1;
557 Repaint ();
558 break;
559 case PathHandle.HandleInputType.LMBRelease:
560 draggingHandleIndex = -1;
561 handleIndexToDisplayAsTransform = -1;
562 Repaint ();
563 break;
564 case PathHandle.HandleInputType.LMBClick:
565 draggingHandleIndex = -1;
566 if (Event.current.shift) {
567 handleIndexToDisplayAsTransform = -1; // disable move tool if new point added
568 } else {
569 if (handleIndexToDisplayAsTransform == i) {
570 handleIndexToDisplayAsTransform = -1; // disable move tool if clicking on point under move tool
571 } else {
572 handleIndexToDisplayAsTransform = i;
573 }
574 }
575 Repaint ();
576 break;
577 case PathHandle.HandleInputType.LMBPress:
578 if (handleIndexToDisplayAsTransform != i) {
579 handleIndexToDisplayAsTransform = -1;
580 Repaint ();
581 }
582 break;
583 }
584
585 Vector3 localHandlePosition = MathUtility.InverseTransformPoint (handlePosition, creator.transform, bezierPath.Space);
586
587 if (bezierPath[i] != localHandlePosition) {
588 Undo.RecordObject (creator, "Move point");
589 bezierPath.MovePoint (i, localHandlePosition);
590
591 }

644 void LoadDisplaySettings () {
645 globalDisplaySettings = GlobalDisplaySettings.Load ();
646
647 capFunctions = new Dictionary<GlobalDisplaySettings.HandleType, Handles.CapFunction> ();
