42 public static Vector3 DrawHandle(Vector3 position, PathSpace space, bool isInteractive, float handleDiameter, Handles.CapFunction capFunc, HandleColours colours, out HandleInputType inputType, int handleIndex)
43 {
44 int id = GetID(handleIndex);
45 Vector3 screenPosition = Handles.matrix.MultiplyPoint(position);
46 Matrix4x4 cachedMatrix = Handles.matrix;
47
48 inputType = HandleInputType.None;
49
50 EventType eventType = Event.current.GetTypeForControl(id);
51 float handleRadius = handleDiameter / 2f;
52 float dstToHandle = HandleUtility.DistanceToCircle(position, handleRadius + extraInputRadius);
53 float dstToMouse = HandleUtility.DistanceToCircle(position, 0);
54
55 // Handle input events
56 if (isInteractive)
57 {
58 // Repaint if mouse is entering/exiting handle (for highlight colour)
59 if (dstToHandle == 0)
60 {
61 if (!mouseIsOverAHandle)
62 {
63 HandleUtility.Repaint();
64 mouseIsOverAHandle = true;
65 }
66 }
67 else
68 {
69 if (mouseIsOverAHandle)
70 {
71 HandleUtility.Repaint();
72 mouseIsOverAHandle = false;
73 }
74 }
75 switch (eventType)
76 {
77 case EventType.MouseDown:
78 if (Event.current.button == 0 && Event.current.modifiers != EventModifiers.Alt)
79 {
80 if (dstToHandle == 0 && dstToMouse < dstMouseToDragPointStart)
81 {
82 dstMouseToDragPointStart = dstToMouse;
83 GUIUtility.hotControl = id;
84 handleDragMouseEnd = handleDragMouseStart = Event.current.mousePosition;
85 handleDragWorldStart = position;
86 selectedHandleID = id;
87 inputType = HandleInputType.LMBPress;
88 }
89 }
90 break;
91
92 case EventType.MouseUp:
93 dstMouseToDragPointStart = float.MaxValue;
94 if (GUIUtility.hotControl == id && Event.current.button == 0)
95 {
96 GUIUtility.hotControl = 0;
97 selectedHandleID = -1;
98 Event.current.Use();
99
100 inputType = HandleInputType.LMBRelease;
101
102
103 if (Event.current.mousePosition == handleDragMouseStart)
104 {
105 inputType = HandleInputType.LMBClick;
106 }
107 }
108 break;
109
110 case EventType.MouseDrag:
111 if (GUIUtility.hotControl == id && Event.current.button == 0)
112 {
113 handleDragMouseEnd += new Vector2(Event.current.delta.x, -Event.current.delta.y);
114 Vector3 position2 = Camera.current.WorldToScreenPoint(Handles.matrix.MultiplyPoint(handleDragWorldStart))
115 + (Vector3)(handleDragMouseEnd - handleDragMouseStart);
116 inputType = HandleInputType.LMBDrag;
117 // Handle can move freely in 3d space
118 if (space == PathSpace.xyz)
119 {
120 position = Handles.matrix.inverse.MultiplyPoint(Camera.current.ScreenToWorldPoint(position2));
121 }
122 // Handle is clamped to xy or xz plane
123 else
124 {
125 position = MouseUtility.GetMouseWorldPosition(space);
126 }
127
128 GUI.changed = true;
129 Event.current.Use();
130 }
131 break;
132 }
133 }
134
135 switch (eventType)
136 {
137 case EventType.Repaint:
138 Color originalColour = Handles.color;
139 Handles.color = (isInteractive) ? colours.defaultColour : colours.disabledColour;
140
141 if (id == GUIUtility.hotControl)
142 {
143 Handles.color = colours.selectedColour;
144 }
145 else if (dstToHandle == 0 && selectedHandleID == -1 && isInteractive)
146 {
147 Handles.color = colours.highlightedColour;
148 }
149
150
151 Handles.matrix = Matrix4x4.identity;
152 Vector3 lookForward = Vector3.up;
153 Camera cam = Camera.current;
154 if (cam != null)
155 {
156 if (cam.orthographic)
157 {
158 lookForward= -cam.transform.forward;
159 }
160 else
161 {
162 lookForward = (cam.transform.position - position);
163 }
164 }
165
166 capFunc(id, screenPosition, Quaternion.LookRotation(lookForward), handleDiameter, EventType.Repaint);
167 Handles.matrix = cachedMatrix;
168
169 Handles.color = originalColour;
170 break;
171
172 case EventType.Layout:
173 Handles.matrix = Matrix4x4.identity;
174 HandleUtility.AddControl(id, HandleUtility.DistanceToCircle(screenPosition, handleDiameter / 2f));
175 Handles.matrix = cachedMatrix;
176 break;
177 }
178
179 return position;
180 }

198 static void AddIDs(int upToIndex)
199 {
200 int numIDAtStart = ids.Count;
201 int numToAdd = (upToIndex - numIDAtStart) + 1;
202 for (int i = 0; i < numToAdd; i++)
203 {
204 string hashString = string.Format("pathhandle({0})", numIDAtStart + i);
205 int hash = hashString.GetHashCode();
206
207 int id = GUIUtility.GetControlID(hash, FocusType.Passive);
208 int numIts = 0;
209
210 // This is a bit of a shot in the dark at fixing a reported bug that I've been unable to reproduce.
211 // The problem is that multiple handles are being selected when just one is clicked on.
212 // I assume this is because they're somehow being assigned the same id.
213 while (idHash.Contains(id)) {
214 numIts ++;
215 id += numIts * numIts;
216 if (numIts > 100) {
217 Debug.LogError("Failed to generate unique handle id.");
218 break;
219 }
220 }
221
222 idHash.Add(id);
223 ids.Add(id);
224 }
225 }
226
227 static int GetID(int handleIndex)
228 {
229 if (handleIndex >= ids.Count)
230 {
231 AddIDs(handleIndex);
232 }
233
234 return ids[handleIndex];
235 }
