35 public ScreenSpacePolyLine(BezierPath bezierPath, Transform transform, float maxAngleError, float minVertexDst, float accuracy = 1)
36 {
37 this.transform = transform;
38 transformPosition = transform.position;
39 transformRotation = transform.rotation;
40 transformScale = transform.localScale;
41
42 // Split path in vertices based on angle error
43 verticesWorld = new List<Vector3>();
44 vertexToPathSegmentMap = new List<int>();
45 segmentStartIndices = new int[bezierPath.NumSegments+1];
46
47 verticesWorld.Add(bezierPath[0]);
48 vertexToPathSegmentMap.Add(0);
49 Vector3 prevPointOnPath = bezierPath[0];
50 float dstSinceLastVertex = 0;
51 Vector3 lastAddedPoint = prevPointOnPath;
52 float dstSinceLastIntermediary = 0;
53
54 for (int segmentIndex = 0; segmentIndex < bezierPath.NumSegments; segmentIndex++)
55 {
56 Vector3[] segmentPoints = bezierPath.GetPointsInSegment(segmentIndex);
57 verticesWorld.Add(segmentPoints[0]);
58 vertexToPathSegmentMap.Add(segmentIndex);
59 segmentStartIndices[segmentIndex] = verticesWorld.Count-1;
60
61 prevPointOnPath = segmentPoints[0];
62 lastAddedPoint = prevPointOnPath;
63 dstSinceLastVertex = 0;
64 dstSinceLastIntermediary = 0;
65
66 float estimatedSegmentLength = CubicBezierUtility.EstimateCurveLength(segmentPoints[0], segmentPoints[1], segmentPoints[2], segmentPoints[3]);
67 int divisions = Mathf.CeilToInt(estimatedSegmentLength * accuracy * accuracyMultiplier);
68 float increment = 1f / divisions;
69
70 for (float t = increment; t <= 1; t += increment)
71 {
72 Vector3 pointOnPath = CubicBezierUtility.EvaluateCurve(segmentPoints[0], segmentPoints[1], segmentPoints[2], segmentPoints[3], t);
73 Vector3 nextPointOnPath = CubicBezierUtility.EvaluateCurve(segmentPoints[0], segmentPoints[1], segmentPoints[2], segmentPoints[3], t + increment);
74
75 // angle at current point on path
76 float localAngle = 180 - MathUtility.MinAngle(prevPointOnPath, pointOnPath, nextPointOnPath);
77 // angle between the last added vertex, the current point on the path, and the next point on the path
78 float angleFromPrevVertex = 180 - MathUtility.MinAngle(lastAddedPoint, pointOnPath, nextPointOnPath);
79 float angleError = Mathf.Max(localAngle, angleFromPrevVertex);
80
81
82 if (angleError > maxAngleError && dstSinceLastVertex >= minVertexDst)
83 {
84 dstSinceLastVertex = 0;
85 dstSinceLastIntermediary = 0;
86 verticesWorld.Add(pointOnPath);
87 vertexToPathSegmentMap.Add(segmentIndex);
88 lastAddedPoint = pointOnPath;
89 }
90 else
91 {
92 if (dstSinceLastIntermediary > intermediaryThreshold)
93 {
94 verticesWorld.Add(pointOnPath);
95 vertexToPathSegmentMap.Add(segmentIndex);
96 dstSinceLastIntermediary = 0;
97 }
98 else
99 {
100 dstSinceLastIntermediary += (pointOnPath - prevPointOnPath).magnitude;
101 }
102 dstSinceLastVertex += (pointOnPath - prevPointOnPath).magnitude;
103 }
104 prevPointOnPath = pointOnPath;
105 }
106 }
107
108 segmentStartIndices[bezierPath.NumSegments] = verticesWorld.Count;
109
110 // ensure final point gets added (unless path is closed loop)
111 if (!bezierPath.IsClosed)
112 {
113 verticesWorld.Add(bezierPath[bezierPath.NumPoints - 1]);
114 }
115 else
116 {
117 verticesWorld.Add(bezierPath[0]);
118 }
119
120 // Calculate length
121 cumululativeLengthWorld = new float[verticesWorld.Count];
122 for (int i = 0; i < verticesWorld.Count; i++)
123 {
124 verticesWorld[i] = MathUtility.TransformPoint(verticesWorld[i], transform, bezierPath.Space);
125 if (i > 0) {
126 pathLengthWorld += (verticesWorld[i - 1] - verticesWorld[i]).magnitude;
127 cumululativeLengthWorld[i] = pathLengthWorld;
128 }
129 }
130
131 }

149 public MouseInfo CalculateMouseInfo()
150 {
151 ComputeScreenSpace();
152
153 Vector2 mousePos = Event.current.mousePosition;
154 float minDst = float.MaxValue;
155 int closestPolyLineSegmentIndex = 0;
156 int closestBezierSegmentIndex = 0;
157
158 for (int i = 0; i < points.Length - 1; i++)
159 {
160 float dst = HandleUtility.DistancePointToLineSegment(mousePos, points[i], points[i + 1]);
161
162 if (dst < minDst)
163 {
164 minDst = dst;
165 closestPolyLineSegmentIndex = i;
166 closestBezierSegmentIndex = vertexToPathSegmentMap[i];
167 }
168 }
169
170 Vector2 closestPointOnLine = MathUtility.ClosestPointOnLineSegment(mousePos, points[closestPolyLineSegmentIndex], points[closestPolyLineSegmentIndex + 1]);
171 float dstToPointOnLine = (points[closestPolyLineSegmentIndex] - closestPointOnLine).magnitude;
172 float percentBetweenVertices = dstToPointOnLine / (points[closestPolyLineSegmentIndex] - points[closestPolyLineSegmentIndex + 1]).magnitude;
173 Vector3 closestPoint3D = Vector3.Lerp(verticesWorld[closestPolyLineSegmentIndex], verticesWorld[closestPolyLineSegmentIndex + 1], percentBetweenVertices);
174
175 float distanceAlongPathWorld = cumululativeLengthWorld[closestPolyLineSegmentIndex] + Vector3.Distance(verticesWorld[closestPolyLineSegmentIndex], closestPoint3D);
176 float timeAlongPath = distanceAlongPathWorld / pathLengthWorld;
177
178 // Calculate how far between the current bezier segment the closest point on the line is
179
180 int bezierSegmentStartIndex = segmentStartIndices[closestBezierSegmentIndex];
181 int bezierSegmentEndIndex = segmentStartIndices[closestBezierSegmentIndex+1];
182 float bezierSegmentLength = cumululativeLengthWorld[bezierSegmentEndIndex] - cumululativeLengthWorld[bezierSegmentStartIndex];
183 float distanceAlongBezierSegment = distanceAlongPathWorld - cumululativeLengthWorld[bezierSegmentStartIndex];
184 float timeAlongBezierSegment = distanceAlongBezierSegment/bezierSegmentLength;
185
186 return new MouseInfo(minDst, closestPoint3D, distanceAlongPathWorld, timeAlongPath, timeAlongBezierSegment, closestBezierSegmentIndex);
187 }
