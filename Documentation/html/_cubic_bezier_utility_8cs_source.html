95 public static List<float> ExtremePointTimes (Vector3 p0, Vector3 p1, Vector3 p2, Vector3 p3) {
96 // coefficients of derivative function
97 Vector3 a = 3 * (-p0 + 3 * p1 - 3 * p2 + p3);
98 Vector3 b = 6 * (p0 - 2 * p1 + p2);
99 Vector3 c = 3 * (p1 - p0);
100
101 List<float> times = new List<float> ();
102 times.AddRange (StationaryPointTimes (a.x, b.x, c.x));
103 times.AddRange (StationaryPointTimes (a.y, b.y, c.y));
104 times.AddRange (StationaryPointTimes (a.z, b.z, c.z));
105 return times;
106 }

108 // Finds times of stationary points on curve defined by ax^2 + bx + c.
109 // Only times between 0 and 1 are considered as Bezier only uses values in that range
110 static IEnumerable<float> StationaryPointTimes (float a, float b, float c) {
111 List<float> times = new List<float> ();
112
113 // from quadratic equation: y = [-b +- sqrt(b^2 - 4ac)]/2a
114 if (a != 0) {
115 float discriminant = b * b - 4 * a * c;
116 if (discriminant >= 0) {
117 float s = Mathf.Sqrt (discriminant);
118 float t1 = (-b + s) / (2 * a);
119 if (t1 >= 0 && t1 <= 1) {
120 times.Add (t1);
121 }
122
123 if (discriminant != 0) {
124 float t2 = (-b - s) / (2 * a);
125
126 if (t2 >= 0 && t2 <= 1) {
127 times.Add (t2);
128 }
129 }
130 }
131 }
132 return times;
133 }
