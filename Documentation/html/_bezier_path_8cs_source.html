228 public void AddSegmentToEnd (Vector3 anchorPos) {
229 if (isClosed) {
230 return;
231 }
232
233 int lastAnchorIndex = points.Count - 1;
234 // Set position for new control to be mirror of its counterpart
235 Vector3 secondControlForOldLastAnchorOffset = (points[lastAnchorIndex] - points[lastAnchorIndex - 1]);
236 if (controlMode != ControlMode.Mirrored && controlMode != ControlMode.Automatic) {
237 // Set position for new control to be aligned with its counterpart, but with a length of half the distance from prev to new anchor
238 float dstPrevToNewAnchor = (points[lastAnchorIndex] - anchorPos).magnitude;
239 secondControlForOldLastAnchorOffset = (points[lastAnchorIndex] - points[lastAnchorIndex - 1]).normalized * dstPrevToNewAnchor * .5f;
240 }
241 Vector3 secondControlForOldLastAnchor = points[lastAnchorIndex] + secondControlForOldLastAnchorOffset;
242 Vector3 controlForNewAnchor = (anchorPos + secondControlForOldLastAnchor) * .5f;
243
244 points.Add (secondControlForOldLastAnchor);
245 points.Add (controlForNewAnchor);
246 points.Add (anchorPos);
247 perAnchorNormalsAngle.Add (perAnchorNormalsAngle[perAnchorNormalsAngle.Count - 1]);
248
249 if (controlMode == ControlMode.Automatic) {
250 AutoSetAllAffectedControlPoints (points.Count - 1);
251 }
252
253 NotifyPathModified ();
254 }

257 public void AddSegmentToStart (Vector3 anchorPos) {
258 if (isClosed) {
259 return;
260 }
261
262 // Set position for new control to be mirror of its counterpart
263 Vector3 secondControlForOldFirstAnchorOffset = (points[0] - points[1]);
264 if (controlMode != ControlMode.Mirrored && controlMode != ControlMode.Automatic) {
265 // Set position for new control to be aligned with its counterpart, but with a length of half the distance from prev to new anchor
266 float dstPrevToNewAnchor = (points[0] - anchorPos).magnitude;
267 secondControlForOldFirstAnchorOffset = secondControlForOldFirstAnchorOffset.normalized * dstPrevToNewAnchor * .5f;
268 }
269
270 Vector3 secondControlForOldFirstAnchor = points[0] + secondControlForOldFirstAnchorOffset;
271 Vector3 controlForNewAnchor = (anchorPos + secondControlForOldFirstAnchor) * .5f;
272 points.Insert (0, anchorPos);
273 points.Insert (1, controlForNewAnchor);
274 points.Insert (2, secondControlForOldFirstAnchor);
275 perAnchorNormalsAngle.Insert (0, perAnchorNormalsAngle[0]);
276
277 if (controlMode == ControlMode.Automatic) {
278 AutoSetAllAffectedControlPoints (0);
279 }
280 NotifyPathModified ();

284 public void SplitSegment (Vector3 anchorPos, int segmentIndex, float splitTime) {
285 splitTime = Mathf.Clamp01 (splitTime);
286
287 if (controlMode == ControlMode.Automatic) {
288 points.InsertRange (segmentIndex * 3 + 2, new Vector3[] { Vector3.zero, anchorPos, Vector3.zero });
289 AutoSetAllAffectedControlPoints (segmentIndex * 3 + 3);
290 } else {
291 // Split the curve to find where control points can be inserted to least affect shape of curve
292 // Curve will probably be deformed slightly since splitTime is only an estimate (for performance reasons, and so doesn't correspond exactly with anchorPos)
293 Vector3[][] splitSegment = CubicBezierUtility.SplitCurve (GetPointsInSegment (segmentIndex), splitTime);
294 points.InsertRange (segmentIndex * 3 + 2, new Vector3[] { splitSegment[0][2], splitSegment[1][0], splitSegment[1][1] });
295 int newAnchorIndex = segmentIndex * 3 + 3;
296 MovePoint (newAnchorIndex - 2, splitSegment[0][1], true);
297 MovePoint (newAnchorIndex + 2, splitSegment[1][2], true);
298 MovePoint (newAnchorIndex, anchorPos, true);
299
300 if (controlMode == ControlMode.Mirrored) {
301 float avgDst = ((splitSegment[0][2] - anchorPos).magnitude + (splitSegment[1][1] - anchorPos).magnitude) / 2;
302 MovePoint (newAnchorIndex + 1, anchorPos + (splitSegment[1][1] - anchorPos).normalized * avgDst, true);
303 }
304 }
305
306 // Insert angle for new anchor (value should be set inbetween neighbour anchor angles)
307 int newAnchorAngleIndex = (segmentIndex + 1) % perAnchorNormalsAngle.Count;
308 int numAngles = perAnchorNormalsAngle.Count;
309 float anglePrev = perAnchorNormalsAngle[segmentIndex];
310 float angleNext = perAnchorNormalsAngle[newAnchorAngleIndex];
311 float splitAngle = Mathf.LerpAngle (anglePrev, angleNext, splitTime);
312 perAnchorNormalsAngle.Insert (newAnchorAngleIndex, splitAngle);
313
314 NotifyPathModified ();
315 }

349 public void MovePoint (int i, Vector3 pointPos, bool suppressPathModifiedEvent = false) {
350
351 if (space == PathSpace.xy) {
352 pointPos.z = 0;
353 } else if (space == PathSpace.xz) {
354 pointPos.y = 0;
355 }
356 Vector3 deltaMove = pointPos - points[i];
357 bool isAnchorPoint = i % 3 == 0;
358
359 // Don't process control point if control mode is set to automatic
360 if (isAnchorPoint || controlMode != ControlMode.Automatic) {
361 points[i] = pointPos;
362
363 if (controlMode == ControlMode.Automatic) {
364 AutoSetAllAffectedControlPoints (i);
365 } else {
366 // Move control points with anchor point
367 if (isAnchorPoint) {
368 if (i + 1 < points.Count || isClosed) {
369 points[LoopIndex (i + 1)] += deltaMove;
370 }
371 if (i - 1 >= 0 || isClosed) {
372 points[LoopIndex (i - 1)] += deltaMove;
373 }
374 }
375 // If not in free control mode, then move attached control point to be aligned/mirrored (depending on mode)
376 else if (controlMode != ControlMode.Free) {
377 bool nextPointIsAnchor = (i + 1) % 3 == 0;
378 int attachedControlIndex = (nextPointIsAnchor) ? i + 2 : i - 2;
379 int anchorIndex = (nextPointIsAnchor) ? i + 1 : i - 1;
380
381 if (attachedControlIndex >= 0 && attachedControlIndex < points.Count || isClosed) {
382 float distanceFromAnchor = 0;
383 // If in aligned mode, then attached control's current distance from anchor point should be maintained
384 if (controlMode == ControlMode.Aligned) {
385 distanceFromAnchor = (points[LoopIndex (anchorIndex)] - points[LoopIndex (attachedControlIndex)]).magnitude;
386 }
387 // If in mirrored mode, then both control points should have the same distance from the anchor point
388 else if (controlMode == ControlMode.Mirrored) {
389 distanceFromAnchor = (points[LoopIndex (anchorIndex)] - points[i]).magnitude;
390
391 }
392 Vector3 dir = (points[LoopIndex (anchorIndex)] - pointPos).normalized;
393 points[LoopIndex (attachedControlIndex)] = points[LoopIndex (anchorIndex)] + dir * distanceFromAnchor;
394 }
395 }
396 }
397
398 if (!suppressPathModifiedEvent) {
399 NotifyPathModified ();
400 }
401 }

405 public Bounds CalculateBoundsWithTransform (Transform transform) {
406 // Loop through all segments and keep track of the minmax points of all their bounding boxes
407 MinMax3D minMax = new MinMax3D ();
408
409 for (int i = 0; i < NumSegments; i++) {
410 Vector3[] p = GetPointsInSegment (i);
411 for (int j = 0; j < p.Length; j++) {
412 p[j] = MathUtility.TransformPoint (p[j], transform, space);
413 }
414
415 minMax.AddValue (p[0]);
416 minMax.AddValue (p[3]);
417
418 List<float> extremePointTimes = CubicBezierUtility.ExtremePointTimes (p[0], p[1], p[2], p[3]);
419 foreach (float t in extremePointTimes) {
420 minMax.AddValue (CubicBezierUtility.EvaluateCurve (p, t));
421 }
422 }
423
424 return new Bounds ((minMax.Min + minMax.Max) / 2, minMax.Max - minMax.Min);
425 }

537 void AutoSetAnchorControlPoints (int anchorIndex) {
538 // Calculate a vector that is perpendicular to the vector bisecting the angle between this anchor and its two immediate neighbours
539 // The control points will be placed along that vector
540 Vector3 anchorPos = points[anchorIndex];
541 Vector3 dir = Vector3.zero;
542 float[] neighbourDistances = new float[2];
543
544 if (anchorIndex - 3 >= 0 || isClosed) {
545 Vector3 offset = points[LoopIndex (anchorIndex - 3)] - anchorPos;
546 dir += offset.normalized;
547 neighbourDistances[0] = offset.magnitude;
548 }
549 if (anchorIndex + 3 >= 0 || isClosed) {
550 Vector3 offset = points[LoopIndex (anchorIndex + 3)] - anchorPos;
551 dir -= offset.normalized;
552 neighbourDistances[1] = -offset.magnitude;
553 }
554
555 dir.Normalize ();
556
557 // Set the control points along the calculated direction, with a distance proportional to the distance to the neighbouring control point
558 for (int i = 0; i < 2; i++) {
559 int controlIndex = anchorIndex + i * 2 - 1;
560 if (controlIndex >= 0 && controlIndex < points.Count || isClosed) {
561 points[LoopIndex (controlIndex)] = anchorPos + dir * neighbourDistances[i] * autoControlLength;
562 }
563 }
564 }

567 void AutoSetStartAndEndControls () {
568 if (isClosed) {
569 // Handle case with only 2 anchor points separately, as will otherwise result in straight line ()
570 if (NumAnchorPoints == 2) {
571 Vector3 dirAnchorAToB = (points[3] - points[0]).normalized;
572 float dstBetweenAnchors = (points[0] - points[3]).magnitude;
573 Vector3 perp = Vector3.Cross (dirAnchorAToB, (space == PathSpace.xy) ? Vector3.forward : Vector3.up);
574 points[1] = points[0] + perp * dstBetweenAnchors / 2f;
575 points[5] = points[0] - perp * dstBetweenAnchors / 2f;
576 points[2] = points[3] + perp * dstBetweenAnchors / 2f;
577 points[4] = points[3] - perp * dstBetweenAnchors / 2f;
578
579 } else {
580 AutoSetAnchorControlPoints (0);
581 AutoSetAnchorControlPoints (points.Count - 3);
582 }
583 } else {
584 // Handle case with 2 anchor points separately, as otherwise minor adjustments cause path to constantly flip
585 if (NumAnchorPoints == 2) {
586 points[1] = points[0] + (points[3] - points[0]) * .25f;
587 points[2] = points[3] + (points[0] - points[3]) * .25f;
588 } else {
589 points[1] = (points[0] + points[2]) * .5f;
590 points[points.Count - 2] = (points[points.Count - 1] + points[points.Count - 3]) * .5f;
591 }
592 }

597 void UpdateToNewPathSpace (PathSpace previousSpace) {
598 // If changing from 3d to 2d space, first find the bounds of the 3d path.
599 // The axis with the smallest bounds will be discarded.
600 if (previousSpace == PathSpace.xyz) {
601 Vector3 boundsSize = PathBounds.size;
602 float minBoundsSize = Mathf.Min (boundsSize.x, boundsSize.y, boundsSize.z);
603
604 for (int i = 0; i < NumPoints; i++) {
605 if (space == PathSpace.xy) {
606 float x = (minBoundsSize == boundsSize.x) ? points[i].z : points[i].x;
607 float y = (minBoundsSize == boundsSize.y) ? points[i].z : points[i].y;
608 points[i] = new Vector3 (x, y, 0);
609 } else if (space == PathSpace.xz) {
610 float x = (minBoundsSize == boundsSize.x) ? points[i].y : points[i].x;
611 float z = (minBoundsSize == boundsSize.z) ? points[i].y : points[i].z;
612 points[i] = new Vector3 (x, 0, z);
613 }
614 }
615 } else {
616 // Nothing needs to change when going to 3d space
617 if (space != PathSpace.xyz) {
618 for (int i = 0; i < NumPoints; i++) {
619 // from xz to xy
620 if (space == PathSpace.xy) {
621 points[i] = new Vector3 (points[i].x, points[i].z, 0);
622 }
623 // from xy to xz
624 else if (space == PathSpace.xz) {
625 points[i] = new Vector3 (points[i].x, 0, points[i].y);
626 }
627 }
628 }
629 }

635 void UpdateClosedState () {
636 if (isClosed) {
637 // Set positions for new controls to mirror their counterparts
638 Vector3 lastAnchorSecondControl = points[points.Count - 1] * 2 - points[points.Count - 2];
639 Vector3 firstAnchorSecondControl = points[0] * 2 - points[1];
640 if (controlMode != ControlMode.Mirrored && controlMode != ControlMode.Automatic) {
641 // Set positions for new controls to be aligned with their counterparts, but with a length of half the distance between start/end anchor
642 float dstBetweenStartAndEndAnchors = (points[points.Count - 1] - points[0]).magnitude;
643 lastAnchorSecondControl = points[points.Count - 1] + (points[points.Count - 1] - points[points.Count - 2]).normalized * dstBetweenStartAndEndAnchors * .5f;
644 firstAnchorSecondControl = points[0] + (points[0] - points[1]).normalized * dstBetweenStartAndEndAnchors * .5f;
645 }
646 points.Add (lastAnchorSecondControl);
647 points.Add (firstAnchorSecondControl);
648 } else {
649 points.RemoveRange (points.Count - 2, 2);
650
651 }
652
653 if (controlMode == ControlMode.Automatic) {
654 AutoSetStartAndEndControls ();
655 }
656
657 if (OnModified != null) {
658 OnModified ();
659 }
