67 public static bool LineSegmentsIntersect (Vector2 a1, Vector2 a2, Vector2 b1, Vector2 b2) {
68 float d = (b2.x - b1.x) * (a1.y - a2.y) - (a1.x - a2.x) * (b2.y - b1.y);
69 if (d == 0)
70 return false;
71 float t = ((b1.y - b2.y) * (a1.x - b1.x) + (b2.x - b1.x) * (a1.y - b1.y)) / d;
72 float u = ((a1.y - a2.y) * (a1.x - b1.x) + (a2.x - a1.x) * (a1.y - b1.y)) / d;
73
74 return t >= 0 && t <= 1 && u >= 0 && u <= 1;
75 }

81 public static Vector2 PointOfLineLineIntersection (Vector2 a1, Vector2 a2, Vector2 a3, Vector2 a4) {
82 float d = (a1.x - a2.x) * (a3.y - a4.y) - (a1.y - a2.y) * (a3.x - a4.x);
83 if (d == 0) {
84 Debug.LogError ("Lines are parallel, please check that this is not the case before calling line intersection method");
85 return Vector2.zero;
86 } else {
87 float n = (a1.x - a3.x) * (a3.y - a4.y) - (a1.y - a3.y) * (a3.x - a4.x);
88 float t = n / d;
89 return a1 + (a2 - a1) * t;

126 public static bool PointInTriangle (Vector2 a, Vector2 b, Vector2 c, Vector2 p) {
127 float area = 0.5f * (-b.y * c.x + a.y * (-b.x + c.x) + a.x * (b.y - c.y) + b.x * c.y);
128 float s = 1 / (2 * area) * (a.y * c.x - a.x * c.y + (c.y - a.y) * p.x + (a.x - c.x) * p.y);
129 float t = 1 / (2 * area) * (a.x * b.y - a.y * b.x + (a.y - b.y) * p.x + (b.x - a.x) * p.y);
130 return s >= 0 && t >= 0 && (s + t) <= 1;
131 }

133 public static bool PointsAreClockwise (Vector2[] points) {
134 float signedArea = 0;
135 for (int i = 0; i < points.Length; i++) {
136 int nextIndex = (i + 1) % points.Length;
137 signedArea += (points[nextIndex].x - points[i].x) * (points[nextIndex].y + points[i].y);
